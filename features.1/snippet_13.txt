# Example usage
import sys

body = HumanBody()
brain = Organ("Brain", mesh_file="example_meshes/brain.obj")
body.add_organ(brain)

# Uncomment to test voxelization
brain.voxelize_mesh()

# Run UI if PyQt5 is available
if 'PyQt5' in globals():
    app = QApplication(sys.argv)
    ui = OrganEditorUI(body)
    ui.show()
    sys.exit(app.exec())
else:
    print("[INFO] PyQt5 not installed, skipping UI.")
    brain.visualize()
